\section{Glossaire}

\begin{enumerate}
	
\item[]{\label{g:segment}\textbf{Segment:} Segment reliant deux stations dans le réseau}
\item[]{\label{g:circuit}\textbf{Circuit:} Ensemble de stations et de segments formant une boucle ou un aller simple dans le réseau}
\item[]{\label{g:besoin}\textbf{Besoin en transport:} Ensemble des besoins pour le déplacement d'un passager}
\item[]{\label{g:gui}\textbf{GUI:} Interface graphique utilisateur du programme}

\end{enumerate}
